The following hotkeys are defined on all pages and can be used to navigate between pages and execute related functions

$(document).bind('keyup', 'left', prevPage);
$(document).bind('keyup', 'right', nextPage);
$(document).bind('keyup', 'f', toggleMargins);
$(document).bind('keyup', 'm', showHideMenu);
$(document).bind('keyup', 'shift+p', prevPage);
$(document).bind('keyup', 'shift+n', nextPage);
